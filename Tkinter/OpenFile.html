Open File

Code to create a button that can be used to browse a file. The file path will be stored in the "HTMLFile" variable. 
def OpenFile():
	global HTMLFile
	file = askopenfile(mode ='r', filetypes =[('HTML Files', '*.html')]) 
	HTMLFile=file.name 

TypeBrowse = Button(root, text ='Open', command = lambda:OpenFile()) 
TypeBrowse.pack(side = TOP, pady = 10) 
TypeBrowse.place(x=110,y=0)
