Open Folder

Code to create a button that can be used to browse a folder. The folder path will be stored in the "filename" variable. 
def browse_button():
	global filename
	filename = filedialog.askdirectory()

TypeBrowse = Button(root, text ='Open', command = browse_button) 
TypeBrowse.pack(side = TOP, pady = 10) 
TypeBrowse.place(x=110,y=0)
